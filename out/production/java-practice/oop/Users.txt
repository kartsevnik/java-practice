package lesson5.tLib.Users;

import java.util.ArrayList;
import java.util.Iterator;

public class Users {

    ArrayList<User> user;

    public Users() {
        user = new ArrayList<User>();
    }

    public User getUser(int u) {
        return user.get(u);
    }

    public void addUser(User u) {
        user.add(u);
    }

    public void removeUser(User u) {
        user.remove(u);
    }

    public void removeUserByIndex(int i) {
        user.remove(i);
    }

    public String toString() {
        Iterator<User> it = user.iterator();
        int counter = 0;
        while (it.hasNext()) {
            User name = it.next();
            System.out.println("#" + counter + ". " + name);
            counter++;
        }
        return null;
    }

}
