package lesson5.tLib.Library;

import java.util.HashMap;

public class Library {

    HashMap<Integer, BookList> library = new HashMap<>();

    public Library() {
        library = new HashMap<Integer, BookList>();
    }

    public BookList getBookList(int i) {
        return library.get(i);
    }

    public void addBookList(int i, BookList bookList) {
        library.put(i, bookList);
    }

    public void removeBookList(int i, BookList bookList) {
        library.remove(i, bookList);
    }

    public String toString() {
        library.forEach((key, value) -> System.out.println("Id: " + key + ", Book: " + value));
        return null;
    }
}
